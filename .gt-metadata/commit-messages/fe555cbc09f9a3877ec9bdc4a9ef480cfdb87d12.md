  var counter = 0

  // Threads
  val thread1 = thread {
    repeat(iterations) {
      counter++
    }
  }

    repeat(iterations) {
      counter++
    }
  }

  // Wait for threads to finish
  thread1.join()
  thread2.join()

  // Display results
  val expected = iterations * 2
  out.infoGreen("Expected value: $expected actual value: $counter.")
  if (counter != expected) {

  }
}
